## instalar dependencias e bibliotecas:
npm install --global yarn
yarn init -y

yarn -i express
yarn add express
yarn add @types/express -D
yarn add typescript -D
yarn tsc --init
yarn add ts-node-dev -D

## iniciar servidor
yarn dev
http://localhost:3333

## para testar metodos POST
instalar aplicação 'insomnia'

## Instalando Banco de Dados
instalar knexjs (knexjs.org)
OU
usar framework typeorm (typeorm.io) ou sequelize
